   240: function haversineKm(lat1, lon1, lat2, lon2){
   241:   const toRad = d => d * Math.PI / 180;
   242:   const R = 6371;
   243:   const dLat = toRad(lat2 - lat1);
   244:   const dLon = toRad(lon2 - lon1);
   245:   const a = Math.sin(dLat / 2) ** 2 + Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLon / 2) ** 2;
   246:   return R * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)));
   247: }
