   199: function haversineKm(lat1, lon1, lat2, lon2){
   200:   const toRad = d => d * Math.PI / 180;
   201:   const R = 6371;
   202:   const dLat = toRad(lat2 - lat1);
   203:   const dLon = toRad(lon2 - lon1);
   204:   const a = Math.sin(dLat / 2) ** 2 +
   205:     Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLon / 2) ** 2;
   206:   return R * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)));
   207: }

   277: async function fetchDriverRoute(fromLat, fromLng, toLat, toLng){
   278:   const la1 = Number(fromLat), ln1 = Number(fromLng), la2 = Number(toLat), ln2 = Number(toLng);
   279:   if(!Number.isFinite(la1) || !Number.isFinite(ln1) || !Number.isFinite(la2) || !Number.isFinite(ln2)){
   280:     return null;
   281:   }
   282:   try{
   283:     const url = `https://router.project-osrm.org/route/v1/driving/${ln1},${la1};${ln2},${la2}?overview=full&geometries=geojson&steps=true`;
   284:     const res = await fetch(url, { cache: "no-store" });
   285:     const json = await res.json().catch(() => null);
   286:     const route = json?.routes?.[0];
   287:     if(!route) return null;
   288:     const coords = Array.isArray(route?.geometry?.coordinates)
   289:       ? route.geometry.coordinates.map((c) => [Number(c?.[1]), Number(c?.[0])]).filter((p) => Number.isFinite(p[0]) && Number.isFinite(p[1]))
   290:       : [];
   291:     const distanceKm = Number(route?.distance || 0) / 1000;
   292:     const durationMin = Number(route?.duration || 0) / 60;
   293:     return {
   294:       path: coords.length >= 2 ? coords : [[la1, ln1], [la2, ln2]],
   295:       distanceKm: Number.isFinite(distanceKm) ? distanceKm : haversineKm(la1, ln1, la2, ln2),
   296:       durationMin: Number.isFinite(durationMin) ? durationMin : Math.max((haversineKm(la1, ln1, la2, ln2) / 24) * 60, 1)
   297:     };
   298:   }catch(_){
   299:     const distanceKm = haversineKm(la1, ln1, la2, ln2);
   300:     return {
   301:       path: [[la1, ln1], [la2, ln2]],
   302:       distanceKm,
   303:       durationMin: Math.max((distanceKm / 24) * 60, 1)
   304:     };
   305:   }
   306: }

   514: function ensureQueueTimer(ride){
   515:   const id = String(ride?.id || "");
   516:   if(!id || state.queueTimers.has(id)) return;
   517:   let remain = 10;
   518:   const interval = setInterval(() => {
   519:     remain -= 1;
   520:     const el = document.getElementById("tm_" + id);
   521:     if(el){
   522:       el.textContent = "Auto reject in " + Math.max(0, remain) + "s";
   523:     }
   524:     if(remain <= 0){
   525:       clearQueueTimer(id);
   526:       rejectRide(id, true).catch(() => {});
   527:     }
   528:   }, 1000);
   529:   state.queueTimers.set(id, { interval });
   530: }
